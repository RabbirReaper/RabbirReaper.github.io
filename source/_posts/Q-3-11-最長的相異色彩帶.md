# 程式碼
```cpp
#include<bits/stdc++.h>
using namespace std;
#define N 200005
int a[N],b[N]={0};
int main(){
    int n,max1=0;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int right=0,left=0,max2=0;right<n;right++){
        b[a[right]]++;
        max2++;
        while(b[a[right]]==2){
            b[a[left]]--;
            left++;
            max2--;
        }
        max1=max(max1,max2);
    }
    cout<<max1;
    return 0;
}
```
