#include <bits/stdc++.h>
using namespace std;
#define Rabbir_Reaper ios::sync_with_stdio(0),cin.tie(0);
int cnt;
void move(char from,char buffer,char to,int disk){
    if(disk==0) return;
    move(from,to,buffer,disk-1);
    cout << "ring " << disk << " : " << from << " => " << to << '\n';
    cnt++;
    move(buffer,from,to,disk-1);
}
int main(){
    Rabbir_Reaper;
    int n;
    char from='A',buffer='B',to='C';
    cin>>n;
    for(int i=n;i>=1;i--){
        move(from,to,buffer,i-1);
        cout << "ring " << i << " : " << from << " => " << to << '\n';
        cnt++;
        i--;
        swap(from,buffer);
        swap(buffer,to);
    }
    cout<<"共需"<<cnt<<"個移動";
    // cout<<cnt;
}
