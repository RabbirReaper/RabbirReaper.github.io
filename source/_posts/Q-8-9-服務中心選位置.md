```cpp
#include<bits/stdc++.h>
using namespace std;
#define N 100005
int total=0,chose[N]={0};
vector<int> path[N];
void dfs(int r,int p){//p = parent
    for(auto e:path[r]){
        if(e==p) continue;
        dfs(e,r);
        if(chose[r]==1)
            chose[e]=1;
        else if(chose[e]==0){
            if(p!=-1 && chose[p]==0) chose[p]=2;
            chose[r]=1;
            chose[e]=1;
            total++;
        }
    }
}
int main(){
    int n,num[N]={1},point=1;
    cin>>n;
    for(int i=1,u,v;i<n;i++){
        cin>>u>>v;
        num[u]++;
        num[v]++;
        path[u].push_back(v);
        path[v].push_back(u);
        if(num[u]>num[point]) point=u;
        if(num[v]>num[point]) point=v;
    }
    dfs(point,-1);
    if(chose[point]==0){
        total++;
    } 
    cout<<total;
    return 0;
}
```
