#include<bits/stdc++.h>
using namespace std;
#define N 26
int a[N],ans,d[N],n;
void sub(int s,int m){
    if(s>n || m<0){
        if(m<ans && m>0)
            ans=m;
        return;
    }
    sub(s+1,m);
    sub(s+1,m-a[s]);
}
int main(){
    int p;
    cin>>n>>p;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    ans=p;
    sub(1,p);
    cout<<p-ans;
    return 0;
}
