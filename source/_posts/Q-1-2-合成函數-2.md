#include<bits/stdc++.h>
using namespace std;
int eval(){
    char a[3];
    int x,y,z;
    cin>>a;
    if(a[0]=='f'){
        x=eval();
        return (2*x-3);
    }else if(a[0]=='g'){
        x=eval();
        y=eval();
        return (2*x+y-7);
    }else if(a[0]=='h'){
        x=eval();
        y=eval();
        z=eval();
        return (3*x-2*y+z);
    }else{
        return atoi(a);
    }
}
int main(){
    cout<<eval();
    return 0;
}
