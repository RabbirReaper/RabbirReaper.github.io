# 程式碼
```cpp
#include<bits/stdc++.h>
using namespace std;
#define N 200005
int b[N]={0},mx=0;
string a[N];
map<string,int> mp;
int main(){
    int m,n,n_color=0;
    cin>>m>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
        mp[a[i]]=0;
    }
    for(auto &i:mp){
        i.second=n_color++;
    }
    for(int right=0,left=0,color;right<n;right++){
        color=mp[a[right]];
        b[color]++;
        while(b[color]==2){
            b[mp[a[left]]]--;
            left++;
        }
        if(right-left+1==m)
            mx++;
    }
    cout<<mx;
    return 0;
}
```
