#include <bits/stdc++.h>
using namespace std;
int a[10];
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int r,c,m;
    vector<vector<int>> ans_v;
    cin>>r>>c>>m;
    for(int i=0;i<r;i++){
        vector<int> temp;
        for(int j=0,t;j<c;j++){
            cin>>t;
            temp.push_back(t);
        }
        ans_v.push_back(temp);
    }
    for(int i=0;i<m;i++) cin>>a[i];
    for(int i=m-1;i>=0;i--){
        if(a[i]==0){
            vector<vector<int>> temp;
            for(int x=(int)ans_v[0].size()-1;x>=0;x--){
                vector<int> buffer;
                for(int y=0;y<(int)ans_v.size();y++){
                    buffer.push_back(ans_v[y][x]);
                }
                temp.push_back(buffer);
            }
            ans_v=temp;
        }else{
            vector<vector<int>> temp;
            for(int x=(int)ans_v.size()-1;x>=0;x--){
                vector<int> buffer;
                for(auto &u:ans_v[x])
                    buffer.push_back(u);
                temp.push_back(buffer);
            }
            ans_v=temp;
        }
    }
    cout<<ans_v.size()<<" "<<ans_v[0].size()<<"\n";
    for(int i=0;i<(int)ans_v.size();i++){
        for(int j=0;j<(int)ans_v[0].size();j++){
            cout<<ans_v[i][j];
            if(j!=(int)ans_v[0].size()-1) cout<<" ";
        }
        if(i!=(int)ans_v.size()-1) cout<<"\n";
    }
}
