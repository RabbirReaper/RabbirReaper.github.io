#include <bits/stdc++.h>
using namespace std;
#define Rabbir_Reaper ios::sync_with_stdio(0);cin.tie();
#define N 400
#define M 800
bool dp[N][M];
int main(){
    Rabbir_Reaper
    int t=2;
    while(t--){
        int n,ans=0;
        cin>>n;
        dp[0][0]=1;
        for(int i=0,t;i<n;i++){
            cin>>t;
            for(int k=N-1;k>=0;k--){
                for(int q=M-1;q>=0;q--) if(dp[k][q]){
                    for(int j=0;j<=t/10;j++){
                        if(k+(t-j*10)/7<N && q+j<M){
                            dp[k+(t-j*10)/7][q+j]=true;
                            ans=max(ans,min(k+(t-j*10)/7,(q+j)/2));
                        }
                    }
                }
            }
        }
        memset(dp,0,sizeof(dp));
        cout<<ans<<"\n";
    }
}
