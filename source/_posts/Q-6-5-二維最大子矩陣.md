# 程式碼
```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
#define N 205
int main(){
    int n,m;
    cin>>n>>m;
    int a[N][N],p[N][N];
    for(int i=0;i<m;i++)
        a[0][i]=0;
    for(int i=1;i<=n;i++){
        for(int j=0;j<m;j++){
            cin>>a[i][j];
            p[i][j]=p[i-1][j]+a[i][j];
        }
    }
    LL mx=0;
    for(int i=0;i<=n;i++){
        for(int j=i+1;j<=n;j++){
            LL sum=0;
            for(int k=0;k<m;k++){
                sum+=p[j][k]-p[i][k];
                if(sum<0) sum=0;
                mx=max(mx,sum);
            }
        }
    }
    cout<<mx;
    return 0;
}
```
