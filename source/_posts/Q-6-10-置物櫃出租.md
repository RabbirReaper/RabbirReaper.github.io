# 程式碼

```cpp
#include<bits/stdc++.h>
using namespace std;
#define N 101
#define W 200005
int dp[N][W]={0};
int main(){
    int n,m,s,a[N],all=0;
    cin>>n>>m>>s;
    for(int i=0;i<n;i++){
        cin>>a[i];
        all+=a[i];
    }
    m-=s;
    for(int i=1;i<=n;i++){
        for(int j=1;j<a[i-1];j++)
            dp[i][j]=dp[i-1][j];//選不了所以跟上一個情況一樣
        for(int j=a[i-1];j<=m;j++)
            dp[i][j]=max(dp[i-1][j],dp[i-1][j-a[i-1]]+a[i-1]);
    }
    cout<<all-dp[n][m];
    return 0;
}
```
