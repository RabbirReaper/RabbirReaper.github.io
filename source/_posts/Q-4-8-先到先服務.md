# 程式碼
```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,m,mx=0;
    priority_queue<int> pq;
    cin>>n>>m;
    for(int i=0,buffer;i<n;i++){
        cin>>buffer;
        if((int)pq.size()<m)
            pq.push(-buffer);
        else{
            buffer+=-(pq.top());
            pq.pop();
            pq.push(-buffer);
        }
        mx=max(mx,buffer);
    }
    cout<<mx;
    return 0;
}
```
