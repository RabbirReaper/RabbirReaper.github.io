# 程式碼
```cpp
#include<bits/stdc++.h>
using namespace std;
#define N 100005
int dp[N]={0};
int main(){
    int n,k,a[N],mx=0;
    cin>>n>>k;
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    dp[0]=a[0];
    for(int i=1;i<=k;i++){
        dp[i]=max(dp[i-1],a[i]);
    }
    for(int i=k+1;i<n;i++){
        dp[i]=max(dp[i-1],a[i]+dp[i-k-1]);
        mx=max(mx,dp[i]);
    }
    cout<<mx;
    return 0;
}
```
